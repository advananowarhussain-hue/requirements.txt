from fastapi import FastAPI, HTTPException, Depends, status
from pydantic import BaseModel, Field
from typing import List, Optional
from motor.motor_asyncio import AsyncIOMotorClient
from bson import ObjectId
import os

app = FastAPI(title="Meriha Hardware API")

# Database Connection (Production-ready via Env)
MONGO_URL = os.getenv("MONGODB_URL", "mongodb://localhost:27017")
client = AsyncIOMotorClient(MONGO_URL)
db = client.meriha_hardware

# Schemas
class CartItem(BaseModel):
    product_id: str
    quantity: int = Field(gt=0, description="Quantity must be greater than 0")

class OrderCreate(BaseModel):
    buyer_name: str
    delivery_address: str
    buyer_phone: str
    payment_method: str  # "COD" or "UPI"
    cart_items: List[CartItem]

@app.post("/api/orders", status_code=status.HTTP_201_CREATED)
async def create_order(order_data: OrderCreate):
    if not order_data.cart_items:
        raise HTTPException(status_code=400, detail="Cart is empty")
        
    calculated_total = 0.0
    verified_items = []
    
    # 1. Server-side Price Validation (Anti-Fraud)
    for item in order_data.cart_items:
        try:
            product = await db.products.find_one({"_id": ObjectId(item.product_id)})
        except:
            raise HTTPException(status_code=400, detail=f"Invalid product ID: {item.product_id}")
            
        if not product:
            raise HTTPException(status_code=404, detail=f"Product not found: {item.product_id}")
            
        # Re-calculating using official DB price
        item_total = float(product["price"]) * item.quantity
        calculated_total += item_total
        
        verified_items.append({
            "product_id": item.product_id,
            "name": product["name"],
            "price": product["price"],
            "quantity": item.quantity,
            "subtotal": item_total
        })
        
    # 2. Prepare Order Document
    new_order = {
        "buyer_name": order_data.buyer_name,
        "delivery_address": order_data.delivery_address,
        "buyer_phone": order_data.buyer_phone,
        "payment_method": order_data.payment_method,
        "items": verified_items,
        "total_amount": calculated_total,
        "status": "Pending",
        "payment_status": "Unpaid" if order_data.payment_method == "UPI" else "COD_Pending"
    }
    
    # 3. Save to MongoDB
    result = await db.orders.insert_one(new_order)
    
    return {
        "success": True,
        "order_id": str(result.inserted_id),
        "total_amount": calculated_total,
        "message": "Order placed successfully!"
